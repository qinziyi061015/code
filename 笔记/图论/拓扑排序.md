拓扑排序，并且按从小到大的顺序赋值
 vector<vector<int>> adj(n+1);
 vector<int> in(n+1,0);
 adj[u].push_back(v);
 in[v]++;
 queue<int> S;
        for(int i=1;i<=n;i++) if(in[i]==0) S.push(i);
        vector<int> L;
        while(!S.empty()){
            int u = S.front(); 
            S.pop();
            L.push_back(u);
            for(int v: adj[u]){
                if(--in[v]==0) S.push(v);
            }
        }
unordered_map<int,int> mp;
        for(int i=0;i<n;i++){
            mp[L[i]] = i+1; 
        }
        for(int i=1;i<=n;i++){
            cout<<mp[i]<<' ';
        }
        cout<<"\n";
